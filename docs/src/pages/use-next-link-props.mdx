# useNextLinkProps

```tsx
"use client";
import { useNextLinkProps } from 'foxact/use-next-link-props'
import Link from 'next/link'

export default function Page() {
  const { isPending, linkProps } = useNextLinkProps({
    href: '/about',
  })
  return (
    <div>
      {isPending && <div>Loading...</div>}
      <Link {...linkProps}>
        About
      </Link>
    </div>
  );
}
```
